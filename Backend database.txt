create database learnersacamedy;
use learnersacamedy;

create table student(  
 id int(10) Not null primary key,  
 name varchar(45) not null,
 age int(3) not null,
 cid int references class(id));
 
 create table class(  
 id int(10) Not null primary key,  
 name varchar(45) not null);

 create table subject(  
 id int(10) Not null primary key,  
 name varchar(45) not null);

 create table teacher(  
 id int(10) Not null primary key,  
 name varchar(45) not null,
 designation varchar(45) not null);


 create table class_subject(cid int references class(id), sid int references subject(id));

 create table teacher_subject(tid int references teacher(id), sid int references subject(id));

 create table teacher_class(tid int references teacher(id), cid int references class(id));
